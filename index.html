function calculateScore() { // Get all the inputs const q1 = document.getElementById('q1').value; const q2 = document.getElementById('q2').value; const q3 = document.getElementById('q3').value; const q4 = document.getElementById('q4').value; const q5 = document.getElementById('q5').value; // Check if everything is answered if (!q1 || !q2 || !q3 || !q4 || !q5) { alert("Hey! Answer all the questions first, please! 🌍"); return; } // Math! const totalScore = parseInt(q1) + parseInt(q2) + parseInt(q3) + parseInt(q4) + parseInt(q5); // Show results document.getElementById('scoreDisplay').innerText = totalScore; const feedback = document.getElementById('feedbackText'); if (totalScore <= 15) { feedback.innerText = "Wow! You are an absolute Planet Hero! 🌿 Keep living that green lifestyle!"; } else if (totalScore <= 35) { feedback.innerText = "You're doing okay, but you could definitely be greener. Small changes make a big difference!"; } else { feedback.innerText = "Oh no! Your footprint is pretty heavy. Check out the facts above and start making changes today! 🚨"; } // Hide form, show result document.getElementById('footprintForm').style.display = 'none'; document.getElementById('resultBox').style.display = 'block'; } function resetQuiz() { document.getElementById('footprintForm').reset(); document.getElementById('footprintForm').style.display = 'block'; document.getElementById('resultBox').style.display = 'none'; }
